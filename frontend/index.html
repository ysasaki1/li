    <script>

      // APIエンドポイント

      const API_BASE_URL = 'http://localhost:3000/api';


      // ポートフォリオを取得する関数

      async function fetchPortfolio() {

        const response = await axios.get(`${API_BASE_URL}/portfolio`);

        const portfolioItems = response.data;

        displayPortfolio(portfolioItems);

      }


      function displayPortfolio(items) {

        const portfolioDiv = document.getElementById('portfolioItems');

        portfolioDiv.innerHTML = ""; // 既存の内容をクリア

        items.forEach(item => {

          const itemDiv = document.createElement('div');

          itemDiv.innerHTML = `

            <h3>${item.title}</h3>

            <p>${item.description}</p>

            <p>寄付金: ${item.donations}円</p>

          `;

          portfolioDiv.appendChild(itemDiv);

        });

      }


      document.getElementById('donationForm').addEventListener('submit', async function(event) {

        event.preventDefault();

        const amount = parseInt(document.getElementById('donationAmount').value, 10);

        if (!isNaN(amount)) {

          // 寄付を送信する

          await axios.post(`${API_BASE_URL}/donate`, { amount });

          fetchPortfolio(); // ポートフォリオを再表示

          document.getElementById('donationAmount').value = ''; // 入力欄をクリア

        }

      });


      // 初期表示

      fetchPortfolio();

    </script>
